package Staff;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileWriter;
import java.io.IOException;
import java.io.PrintWriter;
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class StaffBARepositority implements StaffRepositoryInterface{
    private List<StaffAppointment> appointment;
    private String fileName;
    
    public StaffBARepositority(String fileName){
        this.fileName = fileName;
        this.appointment = loadFromFile();
    }

    public void addAppointment(StaffAppointment a){
        appointment.add(a);
        saveToFile(appointment);
    }

    public List<StaffAppointment> getAllData() {
        return appointment;
    }

    public void saveToFile(List<StaffAppointment> appointment) {
        try (PrintWriter pw = new PrintWriter(new FileWriter(fileName))) {
            for (StaffAppointment a : appointment) {
                pw.println(a.toFileString());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public List<StaffAppointment> loadFromFile() {
        List<StaffAppointment> appointment = new ArrayList<>();
        try (Scanner sc = new Scanner(new File(fileName))) {
            while (sc.hasNextLine()) {
                String line = sc.nextLine().trim();
                if (!line.isEmpty()) {
                    appointment.add(StaffAppointment.fromFileString(line));
                }
            }
        } catch (FileNotFoundException e) {
            System.out.println("No file found, start empty.");
        }
        return appointment;
    }
    
    @Override
    public String generateNextID() {
        String lastId = "B00";
        try (Scanner sc = new Scanner(new File(fileName))) {
            while (sc.hasNextLine()) {
                String line = sc.nextLine().trim();
                if (!line.isEmpty()) {
                    lastId = line.split(",")[0];
                }
            }
        } catch (FileNotFoundException e) {
            System.out.println("File not found, start from B01");
        }
        int num = Integer.parseInt(lastId.substring(1)) + 1;
        return "B" + String.format("%02d", num);
    }
}
